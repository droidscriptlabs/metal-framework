.on

The `.on` function lets you set an event listener that is bound to the element.

When using the '.on' method always stop event propagation.

 this.btn.on('click', Fn)

Here is an example featuring an onclick event:

Example - .on


app.LoadPlugin('Metro')
cfg.Portrait

class Main extends App {
  onStart() {
    this.layMain = ui.addLayout("main", "linear", "fillxy,vcenter");
    this.layMain.setChildMargins(0.02, 0.02, 0.02, 0.02);
    this.txt = ui.addText(this.layMain, "My Hybrid App");

    this.btn = ui.addButton(this.layMain, "My Button", "primary");
    this.btn.on('click', this.onClick)
  }

  onClick(e) {
    // Always stop event propagation
    e.stopPropagation()
    ui.showPopup('You used an event listener')
  } }
