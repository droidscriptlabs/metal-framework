$store

The `$store` function lets you set signals to your variables and subscribe to value changes and set values.

The `$store` function returns a setter/getter and a subscribe function.

 let theme = $store({theme: 'light', lang: 'en'})

Here is an example featuring a theme based signal:

Example - $store


app.LoadPlugin('Metro')
cfg.Portrait

class Main extends App {
  constructor() {
   super()
   this.store = $store({ theme: 'light', lang: 'en'})
   this.store.subscribe(this.onTheme)
  }
  onStart() {
    this.layMain = ui.addLayout("main", "linear", "fillxy,vcenter");
    this.layMain.setChildMargins(0.02, 0.02, 0.02, 0.02);
    this.txt = ui.addText(this.layMain, "My Hybrid App");

    this.btn = ui.addButton(this.layMain, "Switch Theme", "primary");
    this.btn.on('click', this.onClick)
  }

  onTheme(obj) {
    ui.setTheme(obj.theme)   }
  onClick(e) {
    e.stopPropagation()
    if (this.store.get('theme') === 'light') {
      this.store.set('theme', 'dark')
    } else this.store.set('theme', 'light');
  }
}
